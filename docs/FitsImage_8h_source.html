159  void FitsImage<DATA_TYPE>::parse_mask_string(const char *mask_string,
160  long x_resolution,
161  long y_resolution)
162  {
163 
164  const char FITS_MASK_MAX = 0x7F;
165  const char FITS_MASK_DEF = 0x01;
166 
167  if(__mask) delete[] __mask;
168  __mask = new char[x_resolution * y_resolution]();
169 
170  const char *next_term;
171  static int xprev = 0,
172  yprev = 0,
173  use_diff = 0,
174  data = FITS_MASK_DEF;
175 
176  next_term = mask_string;
177  while(*next_term) {
178  int x, y, lx, ly;
179  int num_scanned = sscanf(next_term,
180  "%d,%d:%d,%d",
181  &x, &y, &lx, &ly);
182  if(num_scanned == 0)
183  x = y = lx = ly = 0;
184  else if(num_scanned == 1) {
185  if(x > 0) use_diff = 1;
186  else if(x<0) data=((-x) & FITS_MASK_MAX);
187  else use_diff = 0;
188  x = y = lx = ly = 0;
189  } else if(num_scanned == 2)
190  lx = ly = 1;
191  else if(num_scanned == 3)
192  {
193  if(lx > 1)
194  ly = 1;
195  else if(lx < -1) {
196  ly = -lx;
197  lx = 1;
198  } else
199  lx = ly = 1;
200  }
201 
202  if(lx > 0 && ly > 0)
203  {
204  if(use_diff) {
205  x += xprev;
206  y += yprev;
207  }
208  if(x < 0) {
209  lx += x;
210  x = 0;
211  }
212  if(y < 0) {
213  ly += y;
214  y = 0;
215  }
216  if( x + lx >= static_cast<int>(x_resolution) ) {
217  lx = x_resolution - x;
218  }
219  if( y + ly >= static_cast<int>(y_resolution) ) {
220  ly = y_resolution - y;
221  }
222  xprev = x;
223  yprev = y;
224  for( ; ly > 0 && lx > 0 ; y++, ly-- )
225  {
226  if( y < 0 || y >= static_cast<int>(y_resolution) )
227  continue;
228  char *to_update = (
229  __mask + Core::Image<DATA_TYPE>::index(x, y)
230  );
231  for(int l = 0; l < lx; ++l) {
232  (*to_update) |= data;
233  ++to_update;
234  }
235  }
236  }
237  while (*next_term && *next_term != ' ') ++next_term;
238  while (*next_term == ' ') ++next_term;
239  }
