312  template<class ITERABLE>
313  void EllipticalGaussian::evaluate(double x, double y,
314  ITERABLE result,
315  bool calculate_first_deriv,
316  bool calculate_second_deriv) const
317  {
318  result[NO_DERIV] = exp(exp_argument(x,y));
319  if(calculate_first_deriv || calculate_second_deriv) {
320  double dS_coef = x * x + y * y,
321  dD_coef = x * x - y * y,
322  dK_coef = x * y;
323  if(calculate_first_deriv) {
324  result[S_DERIV] = dS_coef * result[NO_DERIV];
325  result[D_DERIV] = dD_coef * result[NO_DERIV];
326  result[K_DERIV] = dK_coef * result[NO_DERIV];
327  }
328  if(calculate_second_deriv) {
329  result[SS_DERIV] = dS_coef * dS_coef * result[NO_DERIV];
330  result[SD_DERIV] = dS_coef * dD_coef * result[NO_DERIV];
331  result[SK_DERIV] = dS_coef * dK_coef * result[NO_DERIV];
332 
333  result[DD_DERIV] = dD_coef * dD_coef * result[NO_DERIV];
334  result[DK_DERIV] = dD_coef * dK_coef * result[NO_DERIV];
335 
336  result[KK_DERIV] = dK_coef * dK_coef * result[NO_DERIV];
337  }
338  }
339  result[0] += __bg;
340  return;
341  }
