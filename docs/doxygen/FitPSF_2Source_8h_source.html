297  sum_rhs_estimated,
298  sum_sq_rhs);
299  double sum_sq_estimated = estimated_excesses.squaredNorm(),
300  amplitude = sum_rhs_estimated / sum_sq_estimated,
301  result = flux(0).value() - amplitude;
302 
303  flux(0).value() = amplitude;

305  flux(0).error() = std::sqrt(chi2()
306  /
