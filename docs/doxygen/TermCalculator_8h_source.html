637  template<typename Iterator, typename ResultType>
638  void Grammar<Iterator, ResultType>::add_poly_terms(
639  unsigned order,
640  const typename ResultType::const_iterator& first_input_term,
641  const typename ResultType::const_iterator& last_input_term,
642  ResultType &output_terms,
643  const std::string &prefix)
644  {
645  typename ResultType::const_iterator
646  next_input_term=first_input_term;
647  ++next_input_term;
648  if ( next_input_term == last_input_term ) {
649  if ( order == 0 && prefix == "" )
650  output_terms.push_back("1");
651  else {
652  std::ostringstream term_to_add;
653  if ( prefix != "" ) {
654  term_to_add << prefix;
655  if ( order != 0 ) term_to_add << " * ";
656  }
657  if ( order != 0 ) {
658  term_to_add << "(" << *first_input_term << ")";
659  if ( order != 1 )
660  term_to_add << "^" << order;
661  }
662  output_terms.push_back(term_to_add.str());
663  }
664  } else {
665  for ( unsigned remaining_order = 0;
666  remaining_order <= order;
667  ++remaining_order ) {
668  std::ostringstream new_prefix;
669  new_prefix << prefix;
670  if ( remaining_order < order ) {
671  if ( prefix != "" ) new_prefix << " * ";
672  new_prefix << "(" + *first_input_term + ")";
673  if ( order - remaining_order > 1 )
674  new_prefix << "^" << order - remaining_order;
675  }
676  add_poly_terms(remaining_order,
677  next_input_term,
678  last_input_term,
679  output_terms,
680  new_prefix.str());
681  }
682  }
