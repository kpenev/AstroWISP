298  void EllipticalGaussian::evaluate(double x, double y,
299  ITERABLE result,
300  bool calculate_first_deriv,
301  bool calculate_second_deriv) const
302  {
303  result[NO_DERIV] = exp(exp_argument(x,y));
304  if(calculate_first_deriv || calculate_second_deriv) {
305  double dS_coef = x * x + y * y,
306  dD_coef = x * x - y * y,
307  dK_coef = x * y;
308  if(calculate_first_deriv) {
309  result[S_DERIV] = dS_coef * result[NO_DERIV];
310  result[D_DERIV] = dD_coef * result[NO_DERIV];
311  result[K_DERIV] = dK_coef * result[NO_DERIV];
312  }
313  if(calculate_second_deriv) {
314  result[SS_DERIV] = dS_coef * dS_coef * result[NO_DERIV];
315  result[SD_DERIV] = dS_coef * dD_coef * result[NO_DERIV];
316  result[SK_DERIV] = dS_coef * dK_coef * result[NO_DERIV];
317 
318  result[DD_DERIV] = dD_coef * dD_coef * result[NO_DERIV];
319  result[DK_DERIV] = dD_coef * dK_coef * result[NO_DERIV];
320 
321  result[KK_DERIV] = dK_coef * dK_coef * result[NO_DERIV];
322  }
323  }
324  result[0] += __bg;
325  return;
326  }
