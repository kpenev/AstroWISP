297  void EllipticalGaussian::evaluate(double x, double y,
298  ITERABLE result,
299  bool calculate_first_deriv,
300  bool calculate_second_deriv) const
301  {
302  result[NO_DERIV] = exp(exp_argument(x,y));
303  if(calculate_first_deriv || calculate_second_deriv) {
304  double dS_coef = x * x + y * y,
305  dD_coef = x * x - y * y,
306  dK_coef = x * y;
307  if(calculate_first_deriv) {
308  result[S_DERIV] = dS_coef * result[NO_DERIV];
309  result[D_DERIV] = dD_coef * result[NO_DERIV];
310  result[K_DERIV] = dK_coef * result[NO_DERIV];
311  }
312  if(calculate_second_deriv) {
313  result[SS_DERIV] = dS_coef * dS_coef * result[NO_DERIV];
314  result[SD_DERIV] = dS_coef * dD_coef * result[NO_DERIV];
315  result[SK_DERIV] = dS_coef * dK_coef * result[NO_DERIV];
316 
317  result[DD_DERIV] = dD_coef * dD_coef * result[NO_DERIV];
318  result[DK_DERIV] = dD_coef * dK_coef * result[NO_DERIV];
319 
320  result[KK_DERIV] = dK_coef * dK_coef * result[NO_DERIV];
321  }
322  }
323  result[0] += __bg;
324  return;
